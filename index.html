import tkinter as tk from tkinter import messagebox import webbrowser import time class BirthdayApp: def __init__(self, root): self.root = root self.root.title("Birthday Surprise for Dave") self.root.geometry("500x700") self.root.configure(background='#121212') # Tema gelap seperti Spotify self.index = 0 self.messages = ["1", "2", "3", "Happy", "Birthday", "Dave"] # Pesan dari blog sebelumnya (BioSim-Pop) [cite: 24, 25] self.blog_message = ( "Proyek 'BioSim-Pop' adalah langkah nyata integrasi teknologi. " "Semoga harimu penuh dengan eksplorasi dinamis seperti simulasi kita!" ) # Label Utama untuk Animasi Teks self.label = tk.Label(root, text="", font=("Helvetica", 32, "bold"), fg="#1DB954", bg="#121212") self.label.pack(expand=True) # Tombol Mulai self.start_button = tk.Button(root, text="Buka Kejutan", command=self.start_animation, bg="#1DB954", fg="white", font=("Helvetica", 12, "bold")) self.start_button.pack(pady=20) def start_animation(self): """Menjalankan urutan 1, 2, 3, Happy, Birthday, Dave""" self.start_button.destroy() self.show_next_text() def show_next_text(self): if self.index < len(self.messages): self.label.config(text=self.messages[self.index]) self.index += 1 # Kecepatan transisi teks (1 detik) self.root.after(1000, self.show_next_text) else: self.show_final_content() def show_final_content(self): """Menampilkan pesan blog, animasi bunga, dan blok foto""" self.label.destroy() # 1. Animasi Pesan Blog (Pengganti Love) blog_label = tk.Label(self.root, text=self.blog_message, wraplength=400, font=("Courier", 12, "italic"), fg="#FFD700", bg="#121212") blog_label.pack(pady=20) # 2. Representasi Animasi (Bunga/Kue) # Di sini kita menggunakan simbol karakter untuk interaksi sederhana self.canvas = tk.Canvas(self.root, width=200, height=100, bg="#121212", highlightthickness=0) self.canvas.pack() self.flower = self.canvas.create_text(100, 50, text="🌸✨🌷", font=("Helvetica", 40)) self.animate_flower() # 3. Blok Foto & Input Message tk.Label(self.root, text="[ Foto Dave Di Sini ]", bg="grey", width=30, height=10).pack(pady=10) tk.Label(self.root, text="pap paling bener yang pernah kmu kirim ke aku:", fg="white", bg="#121212").pack() self.user_message = tk.Entry(self.root, width=40) self.user_message.pack(pady=5) # 4. Spotify Section (Album Image -> Link) tk.Label(self.root, text="friendship should be fun right..:", fg="white", bg="#121212").pack(pady=10) # Simulasi Gambar Album album_btn = tk.Button(self.root, text="🎵 LIHAT ALBUM 🎵", command=self.open_spotify, bg="#191414", fg="#1DB954") album_btn.pack(pady=5) def animate_flower(self): """Animasi sederhana bunga membesar dan mengecil""" current_font_size = int(self.canvas.itemcget(self.flower, "font").split()[1]) new_size = 45 if current_font_size == 40 else 40 self.canvas.itemconfig(self.flower, font=("Helvetica", new_size)) self.root.after(500, self.animate_flower) def open_spotify(self): """Mengarahkan ke link Spotify""" webbrowser.open("https://open.spotify.com/track/35o9a4iAfLl5jRmqMX9c1D?si=9DKm--9eTZiWvLqSDpnqOQ") # Ganti dengan link playlist Dave messagebox.showinfo("Spotify", "Membuka link Spotify...") if __name__ == "__main__": root = tk.Tk() app = BirthdayApp(root) root.mainloop()
